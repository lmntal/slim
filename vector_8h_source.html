93 template <class T> Vector::Vector(const std::vector<T> &v){
94  static_assert(std::is_scalar<T>::value && sizeof(T) <= sizeof(LmnWord),
95  "vector elements must be scalars.");
96  LMN_ASSERT(v.size() > 0);
97  this->init(v.size());
98  memcpy(this->tbl, v.data(), sizeof(T) * v.size());
99  this->num = v.size();
100 }
