265  void succ_set(Vector *v) {
266  if (!v->is_empty() && !successors) {
267  unsigned int i;
268  successor_num = v->get_num();
269  successors = LMN_NALLOC(succ_data_t, successor_num);
270  for (i = 0; i < successor_num; i++) {
271  successors[i] = (succ_data_t)v->get(i);
272  }
273 #ifdef PROFILE

275  profile_add_space(PROFILE_SPACE__TRANS_OBJECT,
276  sizeof(succ_data_t) * v->get_num());
277  profile_remove_space(PROFILE_SPACE__TRANS_OBJECT, 0);
278  }
279 #endif
280  }

417  /* 状態sに対応する階層グラフ構造Xを,
418  * Xに対して一意なIDとなるバイナリストリングへエンコードする.
419  * エンコードしたバイナリストリングをsに割り当てる.
420  * sに対応する階層グラフ構造Xへのメモリ参照が消えるため,
421  * 呼び出し側でメモリ管理を行う.
422  * sが既にバイナリストリングを保持している場合は,
423  * そのバイナリストリングは破棄する. (ただし,
424  * sが既に一意なIDとなるバイナリストリングへエンコード済みの場合は何もしない.)
425  * 既に割当済みのバイナリストリングを破棄するため,
426  * sをハッシュ表に登録した後の操作はMT-unsafeとなる. 要注意. */
427  void calc_mem_encode() {
428  if (!this->is_encoded()) {
429  LmnBinStrRef mid;
430 
431  if (this->state_mem()) {
432  mid = lmn_mem_encode(this->state_mem());
433  } else if (this->state_binstr()) {
434  LmnMembraneRef m;
435 
436  m = lmn_binstr_decode(this->state_binstr());
437  mid = lmn_mem_encode(m);
438  this->free_binstr();
439  m->free_rec();
440  } else {
441  lmn_fatal("unexpected.");
442  }
443 
444  this->state_set_binstr(mid);
445  this->hash = binstr_hash(this->state_binstr());
446  this->set_encoded();
447  }
448  }

450  void calc_binstr_delta() {
451  LmnBinStrRef org = this->state_binstr();
452  if (org && this->state_D_ref()) {
453  LmnBinStrRef dif;
454  dif = state_binstr_D_compress(org, this->state_D_ref());

462  /* バイナリストリングorgと状態ref_sのバイナリストリングとの差分バイナリストリングを返す.
463  * orgのメモリ管理は呼出し側で行う. */
464  static inline LmnBinStrRef state_binstr_D_compress(LmnBinStrRef org,
465  State *ref_s) {
466  LmnBinStrRef ref, dif;
467 
468  ref = state_D_fetch(ref_s);
469  if (ref) {
470  dif = lmn_bscomp_d_encode(org, ref);
471  } else {
472  ref = ref_s->reconstruct_binstr();
473  dif = lmn_bscomp_d_encode(org, ref);
474  if (ref_s->s_is_d()) {
475  lmn_binstr_free(ref);
476  }
477  }
478 
479  return dif;
480  }

510  /* delta-compression用のinner関数.
511  * flagが真の場合, デコード済みのバイナリストリングをキャッシュから取得する.
512  * キャッシュにバイナリストリングを置かないケースで使用する場合は,
513  * inner関数を直接呼び出し, flagに偽を渡しておけばよい. */
514  LmnMembraneRef restore_membrane_inner(BOOL flag) {
515  if (this->state_mem())
516  return this->state_mem();
517 
518  if (this->s_is_d()) {
519  LmnBinStrRef b =
520  (flag) ? state_D_fetch(this) : this->reconstruct_binstr();
521  return lmn_binstr_decode(b);
522  }
523 
524  LmnBinStrRef b = this->state_binstr();
525  if (lmn_env.tree_compress && !b) {
526  TreeNodeID ref;
527  tcd_get_root_ref(&this->tcd, &ref);
528  LMN_ASSERT(ref);
529  LMN_ASSERT(tcd_get_byte_length(&this->tcd) != 0);
530  b = lmn_bscomp_tree_decode((TreeNodeID)ref,
531  tcd_get_byte_length(&this->tcd));
532  }
533 
534  LMN_ASSERT(b);
535  return lmn_binstr_decode(b);
536  }
