47 /* bsの位置posから1バイト読み込み，返す */
48 static BYTE binstr_get_byte(BYTE *bs, int pos) {
49  return (BS_GET(bs, pos + 1)) | (BS_GET(bs, pos) << 4);
50 }

52 static uint16_t binstr_get_uint16(BYTE *bs, int pos) {
53  return (uint16_t)((binstr_get_byte(bs, pos + 2)) |
54  ((binstr_get_byte(bs, pos)) << 8));
55 }
56 
57 static uint32_t binstr_get_uint32(BYTE *bs, int pos) {
58  return (uint32_t)(binstr_get_uint16(bs, pos + 4) |
59  (binstr_get_uint16(bs, pos) << 16));
60 }
61 
62 static uint64_t binstr_get_uint64(BYTE *bs, int pos) {
63  return (uint64_t)(binstr_get_uint32(bs, pos + 8) |
64  (uint64_t)binstr_get_uint32(bs, pos) << 32);
65 }

80 static double binstr_get_dbl(BYTE *bs, int pos) {
81  int i;
82  union {
83  double d;
84  BYTE t[8];
85  } v;
86 
87  for (i = 7; i >= 0; i--) {
88  v.t[i] = binstr_get_byte(bs, pos + (7 - i) * TAG_IN_BYTE);
89  }
90 
91  return v.d;
92 }

268  std::vector<uint8_t> scan_bytes() {
269  uint64_t size = binstr_get_uint64(bs, index);
270  index += sizeof(uint64_t) * TAG_IN_BYTE;
271  std::vector<uint8_t> bytes(size);
