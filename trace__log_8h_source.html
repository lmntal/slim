67 class LogTracker {
68  std::stack<ProcessID> traced_ids;
69  std::stack<std::size_t> btp_idx;
70 
71 public:
72  void push() { btp_idx.push(traced_ids.size()); }
73 
74  void pop() { btp_idx.pop(); }
75 
76  void trace(ProcessID id) { traced_ids.push(id); }
77 
78  /* 前回のトレースプロセス数を基に, ログを巻き戻す. */
79  template <typename T> void revert(T table) {
80  /* 前回セットしたトレース数のメモを取り出す */
81  const auto bt_idx = btp_idx.top();
82  btp_idx.pop();
83  /* bt_idxより上のidxに積まれたデータをログ上からunputしていく */
84  while (traced_ids.size() > bt_idx) {
85  auto key = traced_ids.top();
86  traced_ids.pop();
87  table->leave(key);
88  }
89  }
90 };

97 /* 各{シンボルアトム, inside proxyアトム, 膜}に対して1つずつ対応させるデータ構造
98  * outside proxyアトムは含めない */
99 struct TraceData { /* 64bit: 24Bytes (32bit: 16Bytes) */
100  BYTE flag; /* 対応させているデータの種類を示すフラグ */
101  unsigned int traversed_proc; /* 膜に対応させている場合は, その膜内で訪問した
102  {シンボルアトム, inside proxyアトム,
103  子膜}の総数 を記録している.
104  他のデータに対応させている場合は0のまま */
105  ProcessID owner_id; /* 対応させているデータ構造の所属膜のID.
106  * プロセスIDは1から始まるため,
107  * 所属膜がない(例えばグローバルルート膜の)場合は, 0 */
108  ProcessID matched; /* 対応させているプロセスとマッチさせたプロセスのID.
109  * in-proxyアトムはBS encode時の訪問順序に数えないため,
110  * in-proxyアトムへの対応としては0をセット */
111 
112  TraceData() : flag(255), traversed_proc(0), owner_id(0), matched(0) {}
113  TraceData(BYTE flag, unsigned int traversed_proc, ProcessID owner_id, ProcessID matched) : flag(flag), traversed_proc(traversed_proc), owner_id(owner_id), matched(matched) {}
114  bool operator==(const TraceData &a) const {
115  return a.flag == flag && a.traversed_proc == traversed_proc &&
116  a.owner_id == owner_id && a.matched == matched;
117  }
118  bool operator!=(const TraceData &a) const { return !(*this == a); }

122  TRAVERSED_MEM = 0x2U,
123  TRAVERSED_HLINK = 0x3U,
124  TRAVERSED_OTHERS = 0xfU,
125  };

128 struct TraceLog {
129  std::map<ProcessID, TraceData> table;
130  LogTracker tracker;
131 
132  using key_type = ProcessTable<TraceData>::key_type;
133 
134  TraceLog(unsigned long size) {}
135  TraceLog() {}
136  unsigned int traversed_proc_count(LmnMembraneRef owner) {
137  return table.find(owner->mem_id()) != table.end()
138  ? table[owner->mem_id()].traversed_proc
139  : 0;
140  }
141 
142 /* 膜ownerを対象として訪問済みにしたプロセス (シンボルアトム + 子膜 + inside
143  * proxies) の数が 膜ownerのそれと一致しているか否かを返す */
144  bool eq_traversed_proc_num(LmnMembraneRef owner, AtomListEntryRef in_ent,
145  AtomListEntryRef avoid) {
146  size_t s1 = in_ent ? in_ent->size() : 0;
147  size_t s2 = avoid ? avoid->size() : 0;
148  return traversed_proc_count(owner) ==
149  (owner->symb_atom_num() + owner->child_mem_num() +
150  s1 - s2);
151  }
152 
153 private:
154  /* ログl上のキーkey位置にあるTraceLogに対して, 訪問を記録する.
155  * matched_idはマッチしたプロセスのID, ownerは所属膜.
156  * 所属膜がNULLでない場合は, 所属膜の情報を記録し,
157  * 所属膜側のプロセス訪問カウンタを回す.
158  */
159  bool visit(LmnWord key, BYTE flag, LmnWord matched_id, LmnMembraneRef owner) {
160  if (table.find(key) != table.end()) {
161  return false;
162  }
163 
164  TraceData value = {flag, 0, 0, matched_id};
165 
166  if (owner) {
167  value.owner_id = owner->mem_id();
168  table[owner->mem_id()].traversed_proc++;
169  }
170 
171  table[key] = value;
172 
173  tracker.trace(key);
174 
175  return true;
176  }
177 
178 public:
179 /* ログに, 所属膜ownerのアトムatomへの訪問を記録する.
180  * atomにマッチしたアトムのプロセスIDもしくは訪問番号atom2_idを併せて記録する.
181  */
182  bool visit_atom(LmnSymbolAtomRef atom, LmnWord atom2_id, LmnMembraneRef owner) {
183  return this->visit(atom->get_id(), TraceData::options::TRAVERSED_ATOM,
184  atom2_id, owner);
185  }
186 
187 /* ログに, 膜mem1への訪問を記録する. (所属膜はmem1のメンバから参照するため不要)
188  * mem1にマッチした膜のプロセスIDもしくは訪問番号mem2_idを併せて記録する */
189  bool visit_mem(LmnMembraneRef mem1, LmnWord mem2_id) {
190  return this->visit(mem1->mem_id(), TraceData::options::TRAVERSED_MEM,
191  mem2_id, mem1->mem_parent());
192  }
193 
194 /* ログに, ハイパーグラフのルートオブジェクトhl1への訪問を記録する.
195  * (ハイパーグラフ構造には所属膜の概念がなく,
196  * 膜オブジェクトからの参照もできないため, 所属膜に対する一切の操作は不要)
197  * hl1にマッチしたハイパリンクオブジェクトIDもしくは訪問番号hl2_idを併せて記録する
198  */
199  bool visit_hlink(HyperLink *hl1, LmnWord hl2_id) {
200  return this->visit(LMN_HL_ID(hl1), TraceData::options::TRAVERSED_HLINK,
201  hl2_id, NULL);
202  }
203 
204  void leave(key_type key) {
205  const auto owner_id = table[key].owner_id;
206  table[owner_id].traversed_proc--;
207  table.erase(key);
208  }
209 
210  bool contains(key_type key) const {
211  return table.find(key) != table.end();
212  }
213 
214  void backtrack() { tracker.revert(this); }
215 
216  void set_btpoint() { tracker.push(); }
217 
218  void continue_trace() { tracker.pop(); }
219 };
