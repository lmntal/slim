206  switch (attr) {

209  case LMN_DBL_ATTR:

212  case LMN_HL_ATTR:
213  return push_hlink(atom, log);
214  case LMN_SP_ATOM_ATTR:

216  auto bytes = sp_atom_encoder(atom)(atom);
217  auto type = LMN_SP_ATOM_TYPE(atom);
218  return push(TAG_SP_ATOM_DATA) && push(&type, sizeof(type) * TAG_IN_BYTE) && push(bytes);

294 #define BS_SET(a, pos, v) \
295  (((pos)&1) \
296  ? ((a)[(pos) >> 1] = ((a)[(pos) >> 1] & 0x0f) | ((v) << TAG_BIT_SIZE)) \
297  : ((a)[(pos) >> 1] = (v & 0x0f) | ((a)[(pos) >> 1] & 0xf0)))

299  (((pos)&1) ? ((a)[(pos) >> 1] & 0xf0) >> TAG_BIT_SIZE \

302 /* エンコード処理(計算中)に用いるバイナリストリング(作業領域) */
303 class BinStr {
304  BYTE *v; /* バイト列(128個で初期化) */
305  int size; /* バッファのサイズ（4ビット単位）: 現在のバイト列の大きさ(128 *
306  TAG_IN_BYTEで初期化) */
307  int cur; /* 書き込み位置（4ビット単位） : 次に書き込む位置(0で初期化) */
308 
309 public:
310  BinStr() {
311  size = BS_TBL_SIZE * TAG_IN_BYTE;
312  v = LMN_NALLOC(BYTE, size / TAG_IN_BYTE);
313  memset(v, 0x0U, sizeof(BYTE) * BS_TBL_SIZE);
314  cur = 0;
315  }
316 
317  ~BinStr() { LMN_FREE(v); }

327 public:
328  /* bsの位置posにbの下位4ビットを書き込む。書き込みに成功した場合は真を
329  返し、失敗した場合は偽を返す。*/
330  bool write(BYTE b, int pos) {
331  resize(pos);
332 
333  /* cur==posならば無条件で書き込み成功 */
334  if (cur == pos) {
335  BS_SET(v, pos, b);
336  cur++;
337  return true;
338  } else if (cur > pos) {
339  /* 下位4ビットの比較を行う */
340  if (BS_GET(v, pos) < b)
341  return false;
342  else if (BS_GET(v, pos) > b) {
343  cur = pos + 1;
344  BS_SET(v, pos, b);
345  return true;
346  } else { /* v[pos] == b */
347  return true;
348  }
349  } else { /*(cur < pos)*/
350  lmn_fatal("unexpected");
351  return false;
352  }
353  }
354 
355  bool write_direct(BYTE b, int pos) {
356  resize(pos);
357 
358  BS_SET(v, pos, b);
359  cur = pos + 1;
360  return true;
361  }

363  LmnBinStr *to_lmn_binstr() const {
364  struct LmnBinStr *ret_bs;
365  int size = (this->cur + 1) / 2;
366  ret_bs = LMN_MALLOC(struct LmnBinStr);
367  ret_bs->v = LMN_NALLOC(BYTE, size);
368  ret_bs->type = 0x00U;
369  memcpy(ret_bs->v, this->v, size);
370  ret_bs->len = this->cur;
371  if (ret_bs->len & 1) {
372  ret_bs->v[ret_bs->len >> 1] = ret_bs->v[ret_bs->len >> 1] & 0x0f;
373  }
374 
375  return ret_bs;
376  }

378 private:
379  void resize(int pos) {
380  while (size <= pos) {
381  int org_size = size / TAG_IN_BYTE;
382  size *= 2;
383  v = LMN_REALLOC(BYTE, v, size / TAG_IN_BYTE);
384  memset(v + org_size, 0x0U,
385  sizeof(BYTE) * ((size / TAG_IN_BYTE) - org_size));
386  }
387  }
