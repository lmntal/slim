59  LmnSymbolAtomRef __next; \
60  if (__ent) { \
61  for ((V) = atomlist_head(__ent); (V) != lmn_atomlist_end(__ent); \
62  (V) = __next) { \
63  __next = (V)->get_next(); \
64  if ((V)->get_functor() != LMN_RESUME_FUNCTOR && \
65  !((D) && dmem_is_removed_atom((D), (MEM), (V)))) { \
66  (CODE); \
67  } \
68  } \
69  } \
70  } while (0)
