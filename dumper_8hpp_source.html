50  void dump_data_atom(halfbyte_scanner &scanner, unsigned int tag) {
51  switch (tag) {
52  case TAG_INT_DATA: {
53  long n = scanner.scan_integer();
54  printf("_INT%ld_ ", n);
55  } break;
56  case TAG_DBL_DATA: {
57  auto n = scanner.scan_double();
58  printf("_DBL%lf_", n);
59  } break;
60  case TAG_SP_ATOM_DATA: {
61  auto type = scanner.scan_sp_atom_type();
62  auto bytes = scanner.scan_bytes();
63  auto atom = sp_atom_decoder(type)(bytes);
64  SP_ATOM_DUMP(atom, lmn_stdout_port());
65  } break;
66  default:
67  lmn_fatal("unexpected.");
68  break;
69  }
70  }

72  void dump(BYTE *bs, int len) {
73  std::vector<BsDecodeLog> log(len * TAG_IN_BYTE);
74  halfbyte_scanner scanner(bs, len);
75  auto v_i = 1;
76  while (scanner.location() < len) {
77  unsigned int tag = scanner.scan_tag();
78 
79  switch (tag) {
80  case TAG_ATOM_START: {
81  LmnFunctor f = scanner.scan_functor();
82  log[v_i] = {(LmnWord)f, BS_LOG_TYPE_ATOM};
83  printf("%s/%d_%d ", lmn_id_to_name(LMN_FUNCTOR_NAME_ID(lmn_functor_table, f)),
84  LMN_FUNCTOR_ARITY(lmn_functor_table, f), v_i++);
85  } break;
86  case TAG_NAMED_MEM_START: /* FALL THROUGH */
87  case TAG_MEM_START: {
88  log[v_i] = {0, BS_LOG_TYPE_MEM}; /* とりえあずゼロクリア */
89 
90  if (tag == TAG_MEM_START) {
91  printf("_%d{ ", v_i);
92  } else {
93  lmn_interned_str name = scanner.scan_mem_name();
94  printf("%s_%d{ ", lmn_id_to_name(name), v_i);
95  }
96 
97  v_i++;
98  } break;
99  case TAG_MEM_END: {
100  printf("}. ");
101  } break;
102  case TAG_HLINK: {
103  LmnHlinkRank hl_num;
104 
105  hl_num = scanner.scan_ref_num();
106 
107  log[v_i] = {0, BS_LOG_TYPE_HLINK}; /* とりあえずゼロクリア */
108  printf("HL/%d_%d ", hl_num, v_i++);
109 
110  tag = scanner.scan_tag();
111  switch (tag) {
112  case TAG_FROM: {
113  printf("_F_ ");
114  } break;
115  case TAG_ATOM_START: {
116  LmnFunctor f = scanner.scan_functor();
117  printf("%s/%d ", lmn_id_to_name(LMN_FUNCTOR_NAME_ID(lmn_functor_table, f)),
118  LMN_FUNCTOR_ARITY(lmn_functor_table, f));
119  } break;
120  case TAG_INT_DATA:
121  case TAG_DBL_DATA:
122  case TAG_SP_ATOM_DATA: {
123  dump_data_atom(scanner, tag);
124  } break;
125  default:
126  lmn_fatal("unexpected");
127  break;
128  }
129  } break;
130  case TAG_VISITED_ATOMHLINK: /* FALL THROUGH */
131  case TAG_VISITED_MEM: {
132  unsigned int ref = scanner.scan_ref_num();
133 
134  switch (log[ref].type) {
135  case BS_LOG_TYPE_ATOM:
136  printf("$%d's%d ", ref, scanner.scan_arg_ref());
137  break;
138  case BS_LOG_TYPE_MEM:
139  printf("#%d ", ref);
140  break;
141  case BS_LOG_TYPE_HLINK:
142  printf("$HL%d ", ref);
143  break;
144  default:
145  lmn_fatal("unexpected reference");
146  break;
147  }
148  } break;
149  case TAG_ESCAPE_MEM_DATA: {
150  unsigned int sub_tag = scanner.scan_tag();
151  printf("!");
152  dump_data_atom(scanner, sub_tag);
153  printf("! ");
154  } break;
155  case TAG_ESCAPE_MEM: {
156  printf("! ");
157  } break;
158  case TAG_FROM: {
159  printf("_F_ ");
160  } break;
161  case TAG_INT_DATA: /* FALL TROUGH */
162  case TAG_DBL_DATA: /* FALL TROUGH */
163  case TAG_SP_ATOM_DATA:
164  dump_data_atom(scanner, tag);
165  break;
166  case TAG_RULESET1: {
167  int rs_id = scanner.scan_ruleset();
168  printf("@%d", rs_id);
169  } break;
170  case TAG_RULESET: {
171  int j, n, rs_id;
172 
173  n = scanner.scan_ruleset_num();
174  for (j = 0; j < n; j++) {
175  rs_id = scanner.scan_ruleset();
176  printf("@%d", rs_id);
177  }
178  } break;
179  case TAG_RULESET_UNIQ: {
180  LmnRuleSetRef rs;
181  lmn_interned_str id;
182  unsigned int j, k, l, n, rs_id, rule_num, his_num;
183 
184  n = scanner.scan_ruleset_num();
185  for (j = 0; j < n; j++) {
186  rs_id = scanner.scan_ruleset();
187  printf("@%d/", rs_id);
188 
189  /* dump applied histories of uniq constraint rules */
190 
191  rs = LmnRuleSetTable::at(rs_id);
192  rule_num = rs->size();
193 
194  for (k = 0; k < rule_num; k++) {
195  printf("[%s", lmn_id_to_name(rs->get_rule(k)->name));
196 
197  his_num = scanner.scan_history_num();
198  for (l = 0; l < his_num; l++) {
199  id = scanner.scan_history();
200  printf("\"%s\"", lmn_id_to_name(id));
201  }
202  printf("]");
203  }
204  }
205  } break;
206  default:
207  printf("pos = %lu, len = %d\n", scanner.location(), len);
208  lmn_fatal("unexpected");
209  break;
210  }
211  }
212  printf("\n");
213  }
